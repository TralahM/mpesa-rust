154#[derive(Debug, Clone, Copy, Serialize)]
155pub enum TransactionType {
156    /// Send Money(Mobile number).
157    SendMoney,
158    /// Withdraw Cash at Agent Till
159    Withdraw,
160    /// Pay Merchant (Buy Goods)
161    BG,
162    /// Paybill or Business number
163    PayBill,
164    /// Sent to Business. Business number CPI in MSISDN format.
165    SendBusiness,
166}
167
168impl Display for TransactionType {
169    fn fmt(&self, f: &mut Formatter) -> FmtResult {
170        write!(f, "{self:?}")
171    }
172}
173
174impl TryFrom<&str> for TransactionType {
175    type Error = MpesaError;
176
177    fn try_from(value: &str) -> Result<Self, Self::Error> {
178        match value.to_lowercase().as_str() {
179            "bg" => Ok(TransactionType::BG),
180            "wa" => Ok(TransactionType::Withdraw),
181            "pb" => Ok(TransactionType::PayBill),
182            "sm" => Ok(TransactionType::SendMoney),
183            "sb" => Ok(TransactionType::SendBusiness),
184            _ => Err(MpesaError::Message("Invalid transaction type")),
185        }
186    }
