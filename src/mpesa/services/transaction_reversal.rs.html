138    /// # Transaction Reversal API
139    ///
140    /// Requests for transaction reversal
141    ///
142    /// This API enables reversal of a B2B, B2C or C2B M-Pesa transaction
143    /// Required  parameters:
144    ///
145    /// `transaction_id`: This is the Mpesa Transaction ID of the transaction which you wish to reverse
146    ///
147    /// `amount` : The amount transacted in the transaction to be reversed , down to the cent
148    ///
149    /// `receiver_party`: Your organization's short code.
150    ///
151    /// See more from the Safaricom API docs [here](https://developer.safaricom.co.ke/Documentation)
152    ///
153    /// A successful request returns a `TransactionReversalResponse` type
154    ///
155    /// # Errors
156    /// Returns a `MpesaError` on failure.
157    pub async fn send(self) -> MpesaResult<TransactionReversalResponse> {
158        self.client
159            .send::<TransactionReversalRequest, _>(crate::client::Request {
160                method: reqwest::Method::POST,
161                path: TRANSACTION_REVERSAL_URL,
162                body: self.try_into()?,
163            })
164            .await
165    }
