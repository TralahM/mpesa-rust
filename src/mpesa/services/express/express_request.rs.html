48    pub amount: u32,

66    /// by your system as an Identifier of the transaction for

69    /// This is any additional information/comment that can be sent along with
70    /// the request from your system
71    pub transaction_desc: Option<&'mpesa str>,

149impl<'mpesa> From<MpesaExpress<'mpesa>> for MpesaExpressRequest<'mpesa> {
150    fn from(express: MpesaExpress<'mpesa>) -> MpesaExpressRequest<'mpesa> {
151        let timestamp = chrono::Local::now();
152
153        let encoded_password = MpesaExpress::encode_password(express.business_short_code, express.pass_key);
154
155        MpesaExpressRequest {
156            business_short_code: express.business_short_code,
157            password: encoded_password,
158            timestamp,
159            transaction_type: express.transaction_type,
160            amount: express.amount,
161            party_a: express.party_a,
162            party_b: express.party_b,
163            phone_number: express.phone_number,
164            call_back_url: express.callback_url,
165            account_reference: express.account_ref,
166            transaction_desc: express.transaction_desc,
167        }
168    }
169}

171impl MpesaExpressBuilder<'_> {
172    /// Validates the request, returning a `MpesaError` if validation fails
173    ///
174    /// Express requests can only be of type `BusinessBuyGoods` or
175    /// `CustomerPayBillOnline`
176    fn validate(&self) -> MpesaResult<()> {
177        if self.transaction_type != Some(CommandId::BusinessBuyGoods)
178            && self.transaction_type != Some(CommandId::CustomerPayBillOnline)
179            && self.transaction_type != Some(CommandId::CustomerBuyGoodsOnline)
180        {
181            return Err(MpesaError::Message(
182                "Invalid transaction type. Expected BusinessBuyGoods or CustomerPayBillOnline or CustomerBuyGoodsOnline",
183            ));
184        }
185
186        if let Some(phone_number) = self.phone_number {
187            phone_number.validate()?;
188        }
189
190        Ok(())
191    }
192}

194impl<'mpesa> MpesaExpress<'mpesa> {
195    /// Creates new `MpesaExpressBuilder`
196    pub(crate) fn builder(client: &'mpesa Mpesa) -> MpesaExpressBuilder<'mpesa> {
197        MpesaExpressBuilder::default().client(client)
198    }
199
200    /// Encodes the password for the request
201    /// The password for encrypting the request is obtained by base64 encoding
202    /// BusinessShortCode, Passkey and Timestamp.
203    /// The timestamp format is YYYYMMDDHHmmss
204    pub fn encode_password(business_short_code: &str, pass_key: Option<&'mpesa str>) -> String {
205        let timestamp = chrono::Local::now().format("%Y%m%d%H%M%S").to_string();
206        encode_block(
207            format!(
208                "{}{}{}",
209                business_short_code,
210                pass_key.unwrap_or(DEFAULT_PASSKEY),
211                timestamp
212            )
213            .as_bytes(),
214        )
215    }
216
217    /// Creates a new `MpesaExpress` from a `MpesaExpressRequest`
218    pub fn from_request(
219        client: &'mpesa Mpesa,
220        request: MpesaExpressRequest<'mpesa>,
221        pass_key: Option<&'mpesa str>,
222    ) -> MpesaExpress<'mpesa> {
223        MpesaExpress {
224            client,
225            business_short_code: request.business_short_code,
226            transaction_type: request.transaction_type,
227            amount: request.amount,
228            party_a: request.party_a,
229            party_b: request.party_b,
230            phone_number: request.phone_number,
231            callback_url: request.call_back_url,
232            account_ref: request.account_reference,
233            transaction_desc: request.transaction_desc,
234            pass_key,
235        }
236    }

238    /// # Lipa na M-Pesa Online Payment / Mpesa Express/ Stk push
239    ///
240    /// Initiates a M-Pesa transaction on behalf of a customer using STK Push
241    ///
242    /// A successful request returns a `MpesaExpressRequestResponse` type
243    ///
244    /// # Errors
245    /// Returns a `MpesaError` on failure
246    pub async fn send(self) -> MpesaResult<MpesaExpressResponse> {
247        self.client
248            .send::<MpesaExpressRequest, _>(crate::client::Request {
249                method: reqwest::Method::POST,
250                path: EXPRESS_REQUEST_URL,
251                body: self.into(),
252            })
253            .await
254    }
