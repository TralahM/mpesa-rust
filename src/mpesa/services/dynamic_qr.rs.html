12#[derive(Debug, Serialize)]
13#[serde(rename_all(serialize = "PascalCase"))]
14pub struct DynamicQRRequest<'mpesa> {
15    /// Name of the Company/M-Pesa Merchant Name
16    pub merchant_name: &'mpesa str,
17    /// Transaction Reference Number
18    pub ref_no: &'mpesa str,
19    /// The total amount of the transaction
20    pub amount: u32,
21    /// Transaction Type
22    ///
23    /// This can be a `TransactionType` or a `&str`
24    /// The `&str` must be one of the following:
25    /// - `BG` for Buy Goods
26    /// - `PB` for Pay Bill
27    /// - `WA` Withdraw Cash
28    /// - `SM` Send Money (Mobile Number)
29    /// - `SB` Sent to Business. Business number CPI in MSISDN format.
30    #[serde(rename = "TrxCode")]
31    pub transaction_type: TransactionType,
32    ///Credit Party Identifier.
33    ///
34    /// Can be a Mobile Number, Business Number, Agent
35    /// Till, Paybill or Business number, or Merchant Buy Goods.
36    #[serde(rename = "CPI")]
37    pub credit_party_identifier: &'mpesa str,
38    /// Size of the QR code image in pixels.
39    ///
40    /// QR code image will always be a square image.
41    pub size: &'mpesa str,
42}

102impl<'mpesa> DynamicQR<'mpesa> {
103    pub(crate) fn builder(client: &'mpesa Mpesa) -> DynamicQRBuilder<'mpesa> {
104        DynamicQRBuilder::default().client(client)
105    }
106
107    /// # Build Dynamic QR
108    ///
109    /// Returns a `DynamicQR` which can be used to send a request
110    pub fn from_request(client: &'mpesa Mpesa, request: DynamicQRRequest<'mpesa>) -> DynamicQR<'mpesa> {
111        DynamicQR {
112            client,
113            merchant_name: request.merchant_name,
114            ref_no: request.ref_no,
115            amount: request.amount,
116            transaction_type: request.transaction_type,
117            credit_party_identifier: request.credit_party_identifier,
118            size: request.size,
119        }
120    }
121
122    /// # Generate a Dynamic QR
123    ///
124    /// This enables Safaricom M-PESA customers who
125    /// have My Safaricom App or M-PESA app, to scan a QR (Quick Response)
126    /// code, to capture till number and amount then authorize to pay for goods
127    /// and services at select LIPA NA M-PESA (LNM) merchant outlets.
128    ///
129    /// # Response
130    /// A successful request returns a `DynamicQRResponse` type
131    /// which contains the QR code
132    ///
133    /// # Errors
134    /// Returns a `MpesaError` on failure
135    pub async fn send(self) -> MpesaResult<DynamicQRResponse> {
136        self.client
137            .send::<DynamicQRRequest, _>(crate::client::Request {
138                method: reqwest::Method::POST,
139                path: DYNAMIC_QR_URL,
140                body: self.into(),
141            })
142            .await
143    }
